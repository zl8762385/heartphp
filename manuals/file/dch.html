DCH

D($model); 获取数据模型

C($filename, $controller = '');多层控制器支持

H($helper); 获取助手工具包

//多层控制器  users = 模版文件名, login = 组名（controller/下文件夹名称）多级别可以往下延伸
$users = C('users', 'login');
$users->test();

//数据模型
$db = D('admin_model');
$db->insert();

//获取助手工具包
$filed = H('helper_content_filed');
$filed->test();

//也可以直接new实例helper_test  和正常操作实例类一样
$h_test = new helper_test();
$h_test->test();
